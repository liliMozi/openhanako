### 环节 6：_handleGroupChat

```javascript
async _handleGroupChat(sessionPath, text, displayMessage) {
  // 1. 文件持久化
  const filePath = "{hanakoHome}/chats/group_chat.jsonl";
  fs.mkdirSync(path.dirname(filePath));

  // 2. 通知前端用户消息已到达
  this._eventBus.emit({type: "session_user_message", message: {text: displayMessage}}, sessionPath);
  this._eventBus.emit({type: "session_status", isStreaming: true}, sessionPath);

  // 3. 写 session JSONL
  fs.promises.appendFile(filePath, [...]);

  // 4. 触发群聊编排
  broadcastToGroup(sessionPath, text, this._groupHistory, {engine, agentIds});
}
```

### 环节 7：broadcastToGroup

```javascript
async broadcastToGroup(groupSessionPath, text, history, {engine, agentIds}) {
  // 1. 保存用户消息到 GroupHistory
  await history.append({id, role:"owner", speaker:"主人", content:text, timestamp});

  // 2. 顺序调用每个 Agent
  const results = [];
  for (const agentId of ["hanako", "mingjian", "suetsuki"]) {
    // 3. 为每个 Agent 创建隔离 Pi SDK session
    const result = await streamAgentToSession(
      agentId, text, history.messages, userMsgId,
      groupSessionPath, {engine, agentDir}
    );
    // 4. Agent 回复写入 GroupHistory
    await history.append({id, role:agentId, speaker:agentId, content:result, timestamp});
    results.push({agentId, success:true, result});
  }
}
```

### 环节 8：streamAgentToSession（核心）

```javascript
async streamAgentToSession(agentId, text, history, excludeId, targetSessionPath, {engine, agentDir}) {
  // 1. 创建临时 SessionManager
  const sessionDir = "{agentDir}/sessions/group_temp_xxx/";
  const tempSessionMgr = SessionManager.create(cwd, sessionDir);

  // 2. 创建完整 Pi SDK session（含工具/模型/人格）
  const {session} = await createAgentSession({
    cwd, sessionManager: tempSessionMgr,
    model, thinkingLevel, tools, customTools,
    resourceLoader: { getSystemPrompt: () => agent.personality },  ← 完整人格
    settingsManager: createDefaultSettings(),
  });

  // 3. 注入群聊历史 + 上下文
  session.agent.state.messages = historyMessages;
  // + 角色映射（hanako/mingjian/suetsuki → "assistant"，owner → "user"）

  // 4. 发送 Agent 状态事件
  engine._emitEvent({type:"agent_status", agentId, status:"started"}, targetSessionPath);
  //                     ↑ 通过 EventBus → chat.js → 广播给所有 WS 客户端

  // 5. 订阅 Pi SDK 事件，转发到 EventBus
  const unsub = session.subscribe((event) => {
    if (event.type === "message_update") {
      const sub = event.assistantMessageEvent;
      if (sub?.type === "text_delta") {
        engine._emitEvent({type:"text_delta", delta: sub.delta}, targetSessionPath);
      } else if (sub?.type === "thinking_delta") {
        engine._emitEvent({type:"thinking_delta", delta: sub.delta}, targetSessionPath);
      }
    } else if (event.type === "turn_end") {
      engine._emitEvent({type:"turn_end"}, targetSessionPath);
    }
  });
  //                     ↑ 通过 EventBus → chat.js hub.subscribe → 处理 → broadcast

  // 6. 调用 LLM（带 180s 超时）
  await Promise.race([session.agent.prompt([currentMessage]), timeout]);

  // 7. 清理
  engine._emitEvent({type:"agent_status", agentId, status:"done"}, targetSessionPath);
  session.dispose(); fs.rmSync(sessionDir, {recursive:true});
}
```

### 环节 9：EventBus 事件 → WS 广播

```
EventBus.emit({type:"text_delta", delta:"大家好"}, "group_chat")

  → toAppEventWsMessage → 非应用事件 → 继续
  → plugin_ui_changed? → 否 → 继续
  → getState("group_chat", event.agentId) → 创建 sessionState（如果不存在）
  → event.type === "text_delta" → ❓ chat.js 处理了 text_delta 吗？
```

### 环节 10：WS 广播 → 前端渲染

```
WS onmessage → handleServerMessage(msg)
  → msg.type === "text_delta"
```

## 四、❗已识别的关键断点

### 🔴 断点 1：事件格式不匹配（R10 当前问题）

**位置**: `group-executor.js:172` → `server/routes/chat.js:237`

**问题**: `streamAgentToSession` 发射 `{type: "text_delta", delta: "..."}` 原始事件。但 `chat.js` 的 `hub.subscribe` handler **只处理 `message_update` 包裹的 text_delta**（L237: `if (event.type === "message_update")`），**不处理裸 `text_delta`**。

`text_delta` 事件到达 `chat.js` 后经过：
```javascript
// chat.js:199-210 → 前两个检查不匹配
// chat.js:212 → getState() → ss 被创建 ✅
// chat.js:237 → if (event.type === "message_update") → ❌ 不匹配！
// chat.js:298 → else if ... 全不匹配
// 事件被静默丢弃，不广播给前端！
```

**正确格式**:
```javascript
engine._emitEvent({
  type: "message_update",                          // ← 外层类型
  assistantMessageEvent: { type: "text_delta", delta: sub.delta }  // ← 内层子类型
}, targetSessionPath);
```

**原因**: 正常 1:1 聊天中，Pi SDK session 所有流式输出都通过 `message_update` 包裹，由 `session-coordinator.js` 统一发射。`group-executor.js` 也监听了 `message_update`（L167），但转发时改成了裸 `text_delta`（R9 修复意图 → 实际制造了新 bug）。
