## 一、三大技术支柱

## ① `ctx.bus` 就是真正的 EventBus

```javascript
// core/plugin-context.js L28-30
const pluginBus = resolvedAccess === "full-access"
  ? bus           // ← 直接引用，不是代理
  : Object.freeze({...});
```

full-access 插件拿到的 `ctx.bus` **就是 Hub 初始化的同一个 EventBus 实例**。这意味着：

- `ctx.bus.request("session:send", { text, sessionPath })` → 直达 Hub handler
- `ctx.bus.subscribe(callback, { types, sessionPath })` → 监听**所有引擎事件**
- `ctx.bus.handle("groupchat:send", handler)` → 注册自己的协议

## ② `session:send` 的 fire-and-forget 特性恰好是优势

```javascript
// hub/index.js L271-282
bus.handle("session:send", async ({ text, sessionPath }) => {
  engine.promptSession(sp, text, opts).catch(err => { ... });
  return { sessionPath: sp, accepted: true };  // ← 立即返回
});
```

它会立即返回 `accepted: true`，然后 Agent 在后台执行。事件（`text_delta`、`tool_start`、`turn_end` 等）通过 EventBus 广播。

**调度器监听 EventBus 的 `turn_end` 来获知 Agent 完成。**

## ③ `extensions/context.js` 注入群聊历史

扩展工厂在 Agent session 创建时被调用，`pi.on()` 注册事件钩子。通过模块级变量把 GroupStore 传进去，在 LLM 调用前注入群聊上下文。

### 4.2 index.js — 调度引擎

```javascript
import path from "path";
import fs from "fs";
import { GroupChatStore } from "./lib/group-store.js";
import { setGroupStore } from "./extensions/context.js";

export default class AgentGroupChatPlugin {
  async onload() {
    const { bus, dataDir, log } = this.ctx;

    // hanakoHome = {hanakoHome}
    const hanakoHome = path.resolve(dataDir, "..", "..");

    // 群聊持久化存储
    const store = new GroupChatStore(dataDir);

    // 共享给 extensions/context.js
    setGroupStore(store);

    // ── groupchat:send — 群聊消息调度 ──
    this.register(bus.handle("groupchat:send", async ({ groupId, text }) => {
      const group = await store.getGroup(groupId);
      if (!group) throw new Error("群聊不存在");

      const userMsgId = Date.now().toString();

      // 1. 写入用户消息
      await store.append(groupId, {
        id: userMsgId,
        role: "owner", speaker: "主人",
        content: text,
        timestamp: Date.now(),
      });

      // 2. 解析 @点名
      let order = [...group.members];
      const mention = text.match(/@(\S+)/);
      if (mention) {
        const nameMap = { "半夏": "hanako", "明鉴": "mingjian", "素月": "suetsuki" };
        const id = nameMap[mention[1]];
        if (id && order.includes(id)) {
          order = [id, ...order.filter(a => a !== id)];
        }
      }

      // 3. 顺序执行每个 Agent
      for (const agentId of order) {
        const sessionPath = path.join(
          hanakoHome, "agents", agentId, "sessions",
          `group_${groupId}.jsonl`,
        );

        // 确保父目录存在
        fs.mkdirSync(path.dirname(sessionPath), { recursive: true });

        log.info(`调度 ${agentId} 回复，session=${sessionPath}`);

        // 触发 Agent 回复
        try {
          await bus.request("session:send", { text, sessionPath });
        } catch (err) {
          log.warn(`${agentId} 启动失败:`, err.message);
          continue;
        }

        // 等待完成并收集回复
        const reply = await waitForAgentDone(bus, sessionPath);

        if (reply.trim()) {
          await store.append(groupId, {
            id: Date.now().toString() + "-" + agentId,
            role: agentId, speaker: agentId,
            content: reply,
            timestamp: Date.now(),
          });
        }
      }
    }));

    // ── groupchat:create — 创建群聊 ──
    this.register(bus.handle("groupchat:create", async ({ name, members }) => {
      const id = `group_${Date.now()}`;
      await store.createGroup({ id, name, members });
      return { id, name, members };
    }));

    // ── groupchat:list — 列出群聊 ──
    this.register(bus.handle("groupchat:list", async () => {
      return store.listGroups();
    }));

    // ── groupchat:messages — 获取消息 ──
    this.register(bus.handle("groupchat:messages", async ({ groupId, since }) => {
      return store.getMessages(groupId, since || 0);
    }));

    log.info("agent-groupchat loaded");
  }
}

/** 等待 Agent 完成回复，收集完整文本 */
function waitForAgentDone(bus, sessionPath, timeoutMs = 180_000) {
  return new Promise((resolve) => {
    let fullText = "";

    const unsub = bus.subscribe((event, sp) => {
      if (sp !== sessionPath) return;

      if (event.type === "message_update") {
        const sub = event.assistantMessageEvent;
        if (sub?.type === "text_delta") {
          fullText += sub.delta || "";
        }
      }

      if (event.type === "turn_end") {
        unsub();
        resolve(fullText);
      }
    });

    setTimeout(() => {
      unsub();
      resolve(fullText);
    }, timeoutMs);
  });
}
```

### 4.3 lib/group-store.js — 持久化

```javascript
import fs from "fs";
import path from "path";

export class GroupChatStore {
  constructor(dataDir) {
    this._dir = path.join(dataDir, "groups");
    fs.mkdirSync(this._dir, { recursive: true });
  }

  _groupPath(groupId) {
    return path.join(this._dir, `${groupId}.jsonl`);
  }

  async createGroup({ id, name, members }) {
    const groupPath = path.join(this._dir, `${id}.json`);
    const group = { id, name, members, createdAt: Date.now() };
    await fs.promises.writeFile(groupPath, JSON.stringify(group, null, 2), "utf-8");
    return group;
  }

  async getGroup(groupId) {
    const groupPath = path.join(this._dir, `${groupId}.json`);
    if (!fs.existsSync(groupPath)) return null;
    return JSON.parse(await fs.promises.readFile(groupPath, "utf-8"));
  }

  listGroups() {
    const groups = [];
    for (const f of fs.readdirSync(this._dir)) {
      if (!f.endsWith(".json")) continue;
      try {
        groups.push(JSON.parse(fs.readFileSync(path.join(this._dir, f), "utf-8")));
      } catch {}
    }
    return groups;
  }

  async append(groupId, message) {
    const line = JSON.stringify(message) + "\n";
    await fs.promises.appendFile(this._groupPath(groupId), line, "utf-8");
  }

  getMessages(groupId, since = 0) {
    const p = this._groupPath(groupId);
    if (!fs.existsSync(p)) return [];
    const lines = fs.readFileSync(p, "utf-8").split("\n").filter(Boolean);
    const messages = [];
    for (const line of lines) {
      try {
        const m = JSON.parse(line);
        if (!since || (m.timestamp && m.timestamp > since)) {
          messages.push(m);
        }
      } catch {}
    }
    return messages;
  }
}
```

### 4.4 extensions/context.js — 上下文注入

```javascript
let _store = null;
export function setGroupStore(store) { _store = store; }

export default function (pi) {
  pi.on("session_start", (event, ctx) => {
    const sessionPath = ctx?.sessionManager?.getSessionFile?.();
    if (!sessionPath || !sessionPath.includes("group_")) return;

    // 从文件名解析: group_{groupId}.jsonl
    const basename = sessionPath.replace(/\\/g, "/").split("/").pop();
    const match = basename.match(/^group_(.+)\.jsonl$/);
    if (!match) return;

    const groupId = match[1];
    const messages = _store?.getMessages(groupId) || [];

    if (messages.length === 0) return;

    // 注入群聊历史到 system prompt
    const history = messages
      .map((m) => `[${m.speaker || m.role}]: ${m.content}`)
      .join("\n");

    const originalSystemPrompt = ctx?.agent?.state?.systemPrompt || "";
    ctx.agent.state.systemPrompt =
      `【群聊 · ${groupId}】
你正在群聊中。以下是最近的对话内容，请以你的角色自然参与。

--- 群聊记录 ---
${history}
---
${originalSystemPrompt}`;
  });
}
```

### 4.5 routes/api.js — HTTP API

```javascript
export default function (app, ctx) {
  const { bus } = ctx;

  app.post("/groupchat/send", async (c) => {
    const { groupId, text } = await c.req.json();
    try {
      await bus.request("groupchat:send", { groupId, text });
      return c.json({ ok: true });
    } catch (err) {
      return c.json({ ok: false, error: err.message }, 400);
    }
  });

  app.get("/groupchat/messages", async (c) => {
    const groupId = c.req.query("groupId");
    const since = parseInt(c.req.query("since") || "0");
    const messages = await bus.request("groupchat:messages", { groupId, since });
    return c.json({ messages });
  });

  app.get("/groupchat/list", async (c) => {
    const groups = await bus.request("groupchat:list", {});
    return c.json({ groups });
  });

  app.post("/groupchat/create", async (c) => {
    const { name, members } = await c.req.json();
    const group = await bus.request("groupchat:create", { name, members });
    return c.json({ group });
  });
}
```
